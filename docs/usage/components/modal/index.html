                        'attributes' => [    'data-bs-dismiss' => 'modal'    ],
                    ],
                ],
            ],
        ],
        [
            'hidden'     => true,
            'fade'       => true,
            'fullscreen' => $size,
            'id'         => 'exampleModalFullscreen'.ucfirst($size),
        ]
    );
}//end foreach
