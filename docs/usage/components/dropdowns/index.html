<?php


// Alignments
foreach ([
    'Dropdown'                       => null,
    'Right-aligned menu'             => 'end',
    'Left-aligned, right-aligned lg' => 'lg-end',
    'Right-aligned, left-aligned lg' => 'lg-start',
] as $label => $alignment) {
    echo $this->formButton()->renderSpec(
        [
            'name'    => 'dropdown',
            'options' => [
                'label'    => $label,
                'dropdown' => [
                    'alignment' => $alignment,
                    'items'     => [
                        'Action',
                        'Another action',
                        'Something else here',
                    ],
                ],
            ],
        ]
    );

    echo PHP_EOL.'<br/>';
}

    'Dropstart' => 'start',
    'Dropend'   => 'end',
    'Dropup'    => 'up',

                ],

                'spec' => [

                        'placeholder' => 'Password',
                    ],
                ],
            ],
            [
                'spec' => [
                    'type'       => 'checkbox',
                    'name'       => 'remember_me',
                    'options'    => [
                        'label'              => 'Remember me',
                        'use_hidden_element' => false,
                    ],
                    'attributes' => [    'id' => 'dropdownCheck'    ],
                ],
            ],
            [
                'spec' => [
                    'type'    => 'submit',

    ]
